Quickstart

Get up and running in minutes.

 

Add a Firebase powered backend to your existing Angular app in 4 easy steps!

  1. Include the Firebase and AngularFire libraries from their respective CDNs.

    <script src="https://cdn.firebase.com/v0/firebase.js"></script>
<script src="https://cdn.firebase.com/libs/angularfire/0.3.0/angularfire.min.js"></script>

  2. Specify a dependency on firebase in your app module.

    var myApp = angular.module("MyApp", ["firebase"]);

  3. Bind a model to a Firebase URL in your controller

    function MyController($scope, angularFire) {
  $scope.items = [];
  angularFire("https://<my-firebase>.firebaseio.com/list", $scope, "items");
}

  4. Use and modify your model as you normally would, watch changes sync in realtime!

    <div ng-controller="MyController">
  <ul ng-repeat="item in items">
    <li>{{item.name}}</li>
  </ul>
  <input type="text" ng-model="newItem"/>
  <button type="submit" ng-click="addItem()">Add New Item</button>
</div>
    function MyController($scope, angularFire) {
  var url = "https://<my-firebase>.firebaseio.com/list";
  $scope.items = [];
  $scope.addItem = function() {
    $scope.items.push($scope.newItem);
    $scope.newItem = "";
  }
  angularFire(url, $scope, "items");
}
